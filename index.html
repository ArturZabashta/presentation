What Is a Regular Expression?

Syntax

The “long” syntax:

							
								const re = new RegExp("pattern", "flags");

								const re = new RegExp("ab+c", 'i');

The “short” syntax:

							
								regexp = /pattern/; // no flags
								const re = /ab+c/;
								
								regexp = /pattern/gmi; // with flags g,m and i
								const re = /ab+c/gmi;

Main difference between the two ways


							
								let regexp = new RegExp(`<${tag}>`);

Patterns

  • Simple Patterns
  • Combinations of common and special characters

Simple Patterns 

							
								console.log("Hi, do you know your abc's?".match(/abc/));  
								//  'abc'								

								console.log("The plane arrived from Cambodia.".match(/abc/));  
								//  null (no match)

Combinations of common and special characters 

							

								console.log("cbbabbbbcdebc".match( /ab*c/));  
								//  'abbbbc'

Character classes


\, ., \cX, \d, \D, \f, \n, \r, \s, \S, \t, \v, \w, \W, \0, \xhh, \uhhhh, \uhhhhh, [\b]

Most used are:

  • . (“.” is “dot”)
  • \d (“d” is from “digit”)
  • \s (“s” is from “space”)
  • \w (“w” is from “word”)


Inverse to them classes

  • \D (non-digit)
  • \S (non-space)
  • \W (non-wordly character)

.

(“.” is “dot”) 
							
								console.log( "Г".match(/./) );     // "Г"

								console.log( "f".match(/./) );     // "f"

								console.log( "8".match(/./) );     // "8"

								console.log( "\n".match(/./) );    // null (no match)

\d

(“d” is from “digit”) 
							
								console.log( "+7(903)-123-45-67".match(/\d/).join('') );  
								// "79031234567"

								console.log( "Hello World".match(/\d/) ); 
								// null (no match)

\s

(“s” is from “space”) 
							
								console.log( "+7(903)-123-45-67".match(/\s/) );  
								// null (no match)

								console.log( "Hello World".match(/\d/) );        
								//  [' ', index: 5, input: 'Hello World', groups: undefined]

\w

(“w” is from “word”) 
							
								console.log( "I love HTML5!".match(/\w\w\w\w/));    
								//  'love'

								console.log( "2022г".match(/\w\w\w\w/));            
								// '2022'
								
								console.log( "2022г".match(/\w\w\w\w\w/));          
								// null (no match)

\D

(non-digit) 
							
								console.log( "+7(903)-123-45-67".replace(/\D/g, "") ); 
								// "79031234567"

								console.log( "2022г".match(/\D/));                     
								// 'г'

\S

(non-space) 
							
								console.log( "\nF1!".match(/\S\S/)); 
								// "F1!"

\W

(non-wordly character) 
							
								console.log( "10000 years назад".match(/\W\W\W\W\W/)); 
								// ' наза'

Flags

RegExp flags in JavaScript:
  • i
  • g
  • m
  • s
  • u
  • y

Flag i :

Enable case-insensitive mode 
															
								let str = "Fee fi fo Fum";								

								let myArray = str.match(/f/); 
								console.log(myArray); /* ['f', index: 4, input: 'Fee fi fo Fum', groups: undefined] */

								let myArrayI = str.match(/f/i);
								console.log(myArrayI); /* ['F', index: 0, input: 'Fee fi fo Fum', groups: undefined] */

Flag g :

Enable search looks for all matches 
							
								let str = "Fee fi fo Fum";								

								let myArray = str.match(/f/); 
								console.log(myArray); 
								/* ['f', index: 4, input: 'Fee fi fo Fum', groups: undefined] */

								let myArrayG = str.match(/f/g); 
								console.log(myArrayG); 
								/* ['f', 'f']  on positions 4 and 7  */

Flag m :

Enable Multiline mode 
							
								let strStart = `1st place: Winnie
								2nd place: Piglet
								3rd place: Eeyore`;
								
								onsole.log( strStart.match(/^\d/g) ); // 1                      

								console.log( strStart.match(/^\d/gm) ); // 1, 2, 3                
								console.log( strStart.match(/^\d/mg) ); // 1, 2, 3                
								
								let strEnd = `Winnie: 1
								Piglet: 2
								Eeyore: 3`;

								console.log( strEnd.match(/\d$/gm) ); // 1,2,3

Flag s :

Enable “dotall” mode 
							
								let str = R\nB;

								console.log( str.match(/R.B/)); // null (no match)     

								console.log( str.match(/R.B/s));  // R\nB (match!)

Flag u :

Enable full Unicode support. 
							
								let  str = `Prices: $2, €1, ¥9`;

								console.log( str.match(/\p{Sc}\d/g) ); 
								/* null (no match)  */
								
								console.log( str.match(/\p{Sc}\d/gu) ); 
								/* $2,€1,¥9  */

Flag y :

Enable “Sticky” mode 
							
								let str = 'let varName = "value"';

								let regexp = /\w+/y;

								regexp.lastIndex = 3;
								alert( regexp.exec(str) ); 
								/* null (there's a space at position 3, not a word) */
								
								regexp.lastIndex = 4;
								alert( regexp.exec(str) );
								/* 'varName' (word at position 4)	*/

Escaping, special characters

  • \
  • [ ]
  • { }
  • ( )
  • ^
  • $
  • |
  • .
  • ?
  • *
  • +
Escaping
							
								console.log( "Chapter 5.1".match(/\d\.\d/) );   
								// "5.1" (match!)

								console.log( "Chapter 511".match(/\d\.\d/) );   
								// null ("\." - loor for a Dot .)

								console.log( "function g()".match(/g\(\)/) );   
								// "g()"

								console.log( "1\\2".match(/\\/) );              
								// "\\"

Working with Regular Expressions

Methods of the RegExp object:
  • regexp.test(str)
  • regexp.exec(str)
Methods of the String object:
  • str.match(regexp)
  • str.matchAll(regexp)
  • str.replace(str|regexp, str|func)
  • str.replaceAll(str|regexp, str|func)
  • str.search(regexp)
  • str.split(regexp|substr, limit)

regexp.test(str)

							
								let str = "I love JavaScript";

								// this two tests do the same

								console.log( /love/.test(str) );           // true

								console.log( str.search(/love/) != -1 );   // true

								console.log( /hate/.test(str) );           // false

regexp.exec(str)

							
								let str = 'Hello, world!';
								let regexp = /\w+/g; // without flag "g", lastIndex property is ignored
								regexp.lastIndex = 5; // search from 5th position (from the comma)
								alert( regexp.exec(str) ); // world								
								
								let regexp = /\w+/y;
								regexp.lastIndex = 5; // search exactly at position 5
								alert( regexp.exec(str) ); // null
